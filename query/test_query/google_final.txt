WITH campaign_criterion_summary AS (
    SELECT
        cc.CAMPAIGN_ID,
        COUNT(*) AS TOTAL_CRITERION_ROWS,
        COUNT_IF(cc.NEGATIVE = FALSE) AS POSITIVE_CRITERION_ROWS,
        COUNT_IF(cc.NEGATIVE = TRUE) AS NEGATIVE_CRITERION_ROWS,
        COUNT_IF(cc.TYPE = 'LOCATION') AS LOCATION_ROWS,
        COUNT_IF(cc.TYPE = 'LOCATION' AND cc.NEGATIVE = FALSE) AS POSITIVE_LOCATION_ROWS,
        COUNT_IF(cc.TYPE = 'LOCATION' AND cc.NEGATIVE = TRUE) AS NEGATIVE_LOCATION_ROWS,
        COUNT_IF(cc.TYPE = 'PROXIMITY') AS PROXIMITY_ROWS,
        COUNT_IF(cc.TYPE = 'LOCATION_GROUP') AS LOCATION_GROUP_ROWS,
        COUNT_IF(cc.TYPE = 'LOCATION_GROUP' AND cc.NEGATIVE = FALSE) AS POSITIVE_LOCATION_GROUP_ROWS,
        COUNT_IF(cc.TYPE = 'LOCATION' AND cc.NEGATIVE = TRUE) AS NEGATIVE_GEO_ROWS,
        LISTAGG(DISTINCT cc.TYPE, ', ')
            WITHIN GROUP (ORDER BY cc.TYPE) AS CRITERION_TYPES,
        LISTAGG(DISTINCT CASE WHEN cc.NEGATIVE = FALSE THEN cc.TYPE END, ', ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.NEGATIVE = FALSE THEN cc.TYPE END) AS POSITIVE_CRITERION_TYPES,
        LISTAGG(DISTINCT CASE WHEN cc.NEGATIVE = TRUE THEN cc.TYPE END, ', ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.NEGATIVE = TRUE THEN cc.TYPE END) AS NEGATIVE_CRITERION_TYPES,
        MAX(cc.OPELAKE_UPDATED_AT) AS CRITERION_UPDATED_AT
    FROM DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.GOOGLE_CAMPAIGN_CRITERION cc
    GROUP BY cc.CAMPAIGN_ID
),
campaign_geo_summary AS (
    SELECT
        cc.CAMPAIGN_ID,
        MAX(CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = FALSE
                  AND geo.TARGET_TYPE = 'Country' THEN 1 ELSE 0 END) AS HAS_COUNTRY,
        MAX(CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = FALSE
                  AND geo.TARGET_TYPE IN ('Province','City','State','District','Municipality',
                                           'Region','Department','Governorate','Prefecture',
                                           'Canton','Borough','Ward','Municipality District',
                                           'Sub District','Sub-District','Sub-Ward',
                                           'Territory','Union Territory','Autonomous Community',
                                           'County','Quarter','Barrio','Commune','Division',
                                           'Okrug','district','neighborhood','Neighborhood')
                 THEN 1 ELSE 0 END) AS HAS_PROVINCE,
        MAX(CASE WHEN cc.TYPE = 'PROXIMITY' THEN 1 ELSE 0 END) AS HAS_PROXIMITY,
        MAX(CASE WHEN cc.TYPE = 'LOCATION_GROUP' AND cc.NEGATIVE = FALSE THEN 1 ELSE 0 END) AS HAS_LOCATION_GROUP,
        MAX(CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = TRUE THEN 1 ELSE 0 END) AS HAS_NEGATIVE_GEO,
        LISTAGG(DISTINCT CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = FALSE THEN geo.TARGET_TYPE END, ', ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = FALSE THEN geo.TARGET_TYPE END) AS LOCATION_TARGET_TYPES,
        LISTAGG(DISTINCT CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = FALSE THEN geo.COUNTRY_CODE END, ', ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = FALSE THEN geo.COUNTRY_CODE END) AS LOCATION_COUNTRY_CODES,
        LISTAGG(DISTINCT CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = FALSE THEN geo.NAME END, ', ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = FALSE THEN geo.NAME END) AS LOCATION_NAMES,
        LISTAGG(DISTINCT CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = FALSE THEN cc.LOCATION_GEO_TARGET_CONSTANT END, ', ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = FALSE THEN cc.LOCATION_GEO_TARGET_CONSTANT END) AS LOCATION_RESOURCE_NAMES,
        LISTAGG(DISTINCT CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = TRUE THEN geo.TARGET_TYPE END, ', ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = TRUE THEN geo.TARGET_TYPE END) AS NEGATIVE_LOCATION_TARGET_TYPES,
        LISTAGG(DISTINCT CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = TRUE THEN geo.COUNTRY_CODE END, ', ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = TRUE THEN geo.COUNTRY_CODE END) AS NEGATIVE_LOCATION_COUNTRY_CODES,
        LISTAGG(DISTINCT CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = TRUE THEN geo.NAME END, ', ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = TRUE THEN geo.NAME END) AS NEGATIVE_LOCATION_NAMES,
        LISTAGG(DISTINCT CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = TRUE THEN cc.LOCATION_GEO_TARGET_CONSTANT END, ', ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.TYPE = 'LOCATION' AND cc.NEGATIVE = TRUE THEN cc.LOCATION_GEO_TARGET_CONSTANT END) AS NEGATIVE_LOCATION_RESOURCE_NAMES,
        MAX(CASE WHEN cc.TYPE = 'PROXIMITY' THEN cc.PROXIMITY_RADIUS END) AS MAX_PROXIMITY_RADIUS,
        MAX(CASE WHEN cc.TYPE = 'PROXIMITY' THEN cc.PROXIMITY_RADIUS_UNITS END) AS PROXIMITY_RADIUS_UNITS,
        MAX(CASE WHEN cc.TYPE = 'LOCATION_GROUP' AND cc.NEGATIVE = FALSE THEN TRY_TO_NUMBER(cc.LOCATION_GROUP:"radius"::VARCHAR) END) AS MAX_LOCATION_GROUP_RADIUS,
        MAX(CASE WHEN cc.TYPE = 'LOCATION_GROUP' AND cc.NEGATIVE = FALSE THEN cc.LOCATION_GROUP:"radiusUnits"::VARCHAR END) AS LOCATION_GROUP_RADIUS_UNITS,
        LISTAGG(DISTINCT CASE WHEN cc.TYPE = 'LOCATION_GROUP' AND cc.NEGATIVE = FALSE THEN TO_VARCHAR(cc.LOCATION_GROUP) END, ' || ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.TYPE = 'LOCATION_GROUP' AND cc.NEGATIVE = FALSE THEN TO_VARCHAR(cc.LOCATION_GROUP) END) AS LOCATION_GROUP_DETAILS,
        LISTAGG(DISTINCT CASE WHEN cc.TYPE = 'PROXIMITY' THEN CONCAT(
            COALESCE(cc.PROXIMITY_ADDRESS_COUNTRY_CODE, ''),
            ':',
            COALESCE(cc.PROXIMITY_ADDRESS_PROVINCE_NAME, ''),
            ':',
            COALESCE(cc.PROXIMITY_ADDRESS_CITY_NAME, ''),
            ':',
            COALESCE(TO_VARCHAR(cc.PROXIMITY_RADIUS), ''),
            ':',
            COALESCE(cc.PROXIMITY_RADIUS_UNITS, '')
        ) END, ' || ')
            WITHIN GROUP (ORDER BY CASE WHEN cc.TYPE = 'PROXIMITY' THEN CONCAT(
                COALESCE(cc.PROXIMITY_ADDRESS_COUNTRY_CODE, ''),
                ':',
                COALESCE(cc.PROXIMITY_ADDRESS_PROVINCE_NAME, ''),
                ':',
                COALESCE(cc.PROXIMITY_ADDRESS_CITY_NAME, ''),
                ':',
                COALESCE(TO_VARCHAR(cc.PROXIMITY_RADIUS), ''),
                ':',
                COALESCE(cc.PROXIMITY_RADIUS_UNITS, '')
            ) END) AS PROXIMITY_DETAILS
    FROM DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.GOOGLE_CAMPAIGN_CRITERION cc
    LEFT JOIN DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.GOOGLE_GEO_TARGET_CONSTANT geo
        ON cc.LOCATION_GEO_TARGET_CONSTANT = geo.RESOURCE_NAME
    GROUP BY cc.CAMPAIGN_ID
)
SELECT
    c.CUSTOMER_ID,
    cust.DESCRIPTIVE_NAME AS CUSTOMER_NAME,
    c.ID AS CAMPAIGN_ID,
    c.NAME AS CAMPAIGN_NAME,
    c.STATUS,
    c.ADVERTISING_CHANNEL_TYPE,
    CASE
        WHEN c.ADVERTISING_CHANNEL_TYPE = 'MULTI_CHANNEL' THEN 'GAC'
        WHEN c.ADVERTISING_CHANNEL_TYPE = 'DISPLAY' THEN 'Google_GDN'
        WHEN c.ADVERTISING_CHANNEL_TYPE = 'VIDEO' THEN 'YouTube'
        WHEN c.ADVERTISING_CHANNEL_TYPE = 'LOCAL' THEN 'O2O'
        ELSE c.ADVERTISING_CHANNEL_TYPE
    END AS MEDIA,
    c.OPELAKE_UPDATED_AT AS CAMPAIGN_UPDATED_AT,
    c.GEO_TARGET_TYPE_SETTING_POSITIVE_GEO_TARGET_TYPE,
    c.GEO_TARGET_TYPE_SETTING_NEGATIVE_GEO_TARGET_TYPE,
    CASE
        WHEN c.GEO_TARGET_TYPE_SETTING_POSITIVE_GEO_TARGET_TYPE IS NOT NULL
          OR c.GEO_TARGET_TYPE_SETTING_NEGATIVE_GEO_TARGET_TYPE IS NOT NULL THEN 1
        ELSE 0
    END AS HAS_GEO_MODE_SETTING,
    COALESCE(cs.TOTAL_CRITERION_ROWS, 0) AS TOTAL_CRITERION_ROWS,
    COALESCE(cs.POSITIVE_CRITERION_ROWS, 0) AS POSITIVE_CRITERION_ROWS,
    COALESCE(cs.NEGATIVE_CRITERION_ROWS, 0) AS NEGATIVE_CRITERION_ROWS,
    COALESCE(cs.LOCATION_ROWS, 0) AS LOCATION_ROWS,
    COALESCE(cs.POSITIVE_LOCATION_ROWS, 0) AS POSITIVE_LOCATION_ROWS,
    COALESCE(cs.NEGATIVE_LOCATION_ROWS, 0) AS NEGATIVE_LOCATION_ROWS,
    COALESCE(cs.PROXIMITY_ROWS, 0) AS PROXIMITY_ROWS,
    COALESCE(cs.LOCATION_GROUP_ROWS, 0) AS LOCATION_GROUP_ROWS,
    COALESCE(cs.POSITIVE_LOCATION_GROUP_ROWS, 0) AS POSITIVE_LOCATION_GROUP_ROWS,
    COALESCE(cs.NEGATIVE_GEO_ROWS, 0) AS NEGATIVE_GEO_ROWS,
    cs.CRITERION_TYPES,
    cs.POSITIVE_CRITERION_TYPES,
    cs.NEGATIVE_CRITERION_TYPES,
    cs.CRITERION_UPDATED_AT,
    COALESCE(gs.HAS_COUNTRY, 0) AS HAS_COUNTRY,
    COALESCE(gs.HAS_PROVINCE, 0) AS HAS_PROVINCE,
    COALESCE(gs.HAS_PROXIMITY, 0) AS HAS_PROXIMITY,
    COALESCE(gs.HAS_LOCATION_GROUP, 0) AS HAS_LOCATION_GROUP,
    COALESCE(gs.HAS_NEGATIVE_GEO, 0) AS HAS_NEGATIVE_GEO,
    gs.LOCATION_TARGET_TYPES,
    gs.LOCATION_COUNTRY_CODES,
    gs.LOCATION_NAMES,
    gs.LOCATION_RESOURCE_NAMES,
    gs.NEGATIVE_LOCATION_TARGET_TYPES,
    gs.NEGATIVE_LOCATION_COUNTRY_CODES,
    gs.NEGATIVE_LOCATION_NAMES,
    gs.NEGATIVE_LOCATION_RESOURCE_NAMES,
    gs.MAX_PROXIMITY_RADIUS,
    gs.PROXIMITY_RADIUS_UNITS,
    gs.MAX_LOCATION_GROUP_RADIUS,
    gs.LOCATION_GROUP_RADIUS_UNITS,
    gs.PROXIMITY_DETAILS,
    gs.LOCATION_GROUP_DETAILS,
    CASE
        WHEN COALESCE(cs.TOTAL_CRITERION_ROWS, 0) = 0
          AND (c.GEO_TARGET_TYPE_SETTING_POSITIVE_GEO_TARGET_TYPE IS NOT NULL
            OR c.GEO_TARGET_TYPE_SETTING_NEGATIVE_GEO_TARGET_TYPE IS NOT NULL) THEN 'HAS_GEO_MODE_SETTING_ONLY'
        WHEN COALESCE(cs.TOTAL_CRITERION_ROWS, 0) = 0 THEN 'NO_CRITERION_DATA'
        WHEN COALESCE(gs.HAS_COUNTRY, 0) = 1
          OR COALESCE(gs.HAS_PROVINCE, 0) = 1
          OR COALESCE(gs.HAS_PROXIMITY, 0) = 1
          OR COALESCE(gs.HAS_LOCATION_GROUP, 0) = 1 THEN 'HAS_GEO_TARGETING'
        WHEN COALESCE(gs.HAS_NEGATIVE_GEO, 0) = 1 THEN 'HAS_NEGATIVE_GEO_ONLY'
        ELSE 'NO_GEO_TARGETING'
    END AS DATA_COMPLETENESS_STATUS,
    CASE
        WHEN COALESCE(cs.TOTAL_CRITERION_ROWS, 0) = 0
          AND (c.GEO_TARGET_TYPE_SETTING_POSITIVE_GEO_TARGET_TYPE IS NOT NULL
            OR c.GEO_TARGET_TYPE_SETTING_NEGATIVE_GEO_TARGET_TYPE IS NOT NULL) THEN 'SKIP'
        WHEN COALESCE(cs.TOTAL_CRITERION_ROWS, 0) = 0 THEN 'REVIEW_ETL'
        WHEN COALESCE(gs.HAS_COUNTRY, 0) = 1
          OR COALESCE(gs.HAS_PROVINCE, 0) = 1
          OR COALESCE(gs.HAS_PROXIMITY, 0) = 1
          OR COALESCE(gs.HAS_LOCATION_GROUP, 0) = 1
          OR COALESCE(gs.HAS_NEGATIVE_GEO, 0) = 1 THEN 'SKIP'
        ELSE 'GET'
    END AS DECISION
FROM DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.GOOGLE_CAMPAIGN c
LEFT JOIN DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.GOOGLE_CUSTOMER cust
    ON c.CUSTOMER_ID = cust.ID
LEFT JOIN campaign_criterion_summary cs
    ON cs.CAMPAIGN_ID = c.ID
LEFT JOIN campaign_geo_summary gs
    ON gs.CAMPAIGN_ID = c.ID
WHERE c.ADVERTISING_CHANNEL_TYPE IN ('MULTI_CHANNEL', 'DISPLAY', 'VIDEO', 'LOCAL')
    AND c.STATUS != 'REMOVED'
ORDER BY MEDIA, DATA_COMPLETENESS_STATUS, DECISION, c.NAME

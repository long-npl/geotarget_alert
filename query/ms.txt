WITH bing_geo_flags AS (
    SELECT
        CAMPAIGN_ID,
        MAX(CASE WHEN TYPE = 'Campaign Location Criterion' AND SUB_TYPE = 'Country' THEN 1 ELSE 0 END) AS HAS_COUNTRY,
        MAX(CASE WHEN TYPE = 'Campaign Location Criterion'
                  AND SUB_TYPE IN ('State','City','Neighborhood','MetroArea','PostalCode')
                 THEN 1 ELSE 0 END) AS HAS_PROVINCE,
        MAX(CASE WHEN TYPE = 'Campaign Radius Criterion' THEN 1 ELSE 0 END) AS HAS_PROXIMITY
    FROM DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.BING_CAMPAIGN_TARGET_CRITERION
    GROUP BY CAMPAIGN_ID
)
SELECT
    c.ACCOUNT_ID,
    a.NAME                AS ACCOUNT_NAME,
    a.NUMBER              AS ACCOUNT_NUMBER,
    c.ID                  AS CAMPAIGN_ID,
    c.NAME                AS CAMPAIGN_NAME,
    c.STATUS,
    c.CAMPAIGN_TYPE       AS MEDIA,
    cc.TARGET             AS COUNTRY_TARGET
FROM DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.BING_CAMPAIGN c
LEFT JOIN DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.BING_ACCOUNT a
    ON c.ACCOUNT_ID = a.ID
LEFT JOIN DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.BING_CAMPAIGN_TARGET_CRITERION cc
    ON cc.CAMPAIGN_ID = c.ID
   AND cc.TYPE = 'Campaign Location Criterion'
   AND cc.SUB_TYPE = 'Country'
LEFT JOIN bing_geo_flags f
    ON f.CAMPAIGN_ID = c.ID
WHERE cc.TARGET IS NULL
    AND COALESCE(f.HAS_COUNTRY,0) = 0
    AND COALESCE(f.HAS_PROVINCE,0) = 0
    AND COALESCE(f.HAS_PROXIMITY,0) = 0
ORDER BY MEDIA, c.NAME;

SELECT
    s.ACCOUNT_ID,
    aa.NAME AS ACCOUNT_NAME,
    s.CAMPAIGN_ID,
    ac.NAME AS CAMPAIGN_NAME,
    s.ID AS AD_SET_ID,
    s.NAME AS AD_SET_NAME,
    s.EFFECTIVE_STATUS,
    s.TARGETING:geo_locations:country_groups AS GLOBAL_REGIONS,
    s.TARGETING:geo_locations:location_types AS LOCATION_TYPES
FROM DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.FACEBOOK_AD_SET s
LEFT JOIN DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.FACEBOOK_AD_ACCOUNT aa
    ON s.ACCOUNT_ID = aa.ACCOUNT_ID
LEFT JOIN DM_GEOTARGET_ALERT_PRD.DATAMART_SCHEMA.FACEBOOK_AD_CAMPAIGN ac
    ON s.CAMPAIGN_ID = ac.ID
WHERE s.TARGETING:geo_locations:country_groups::VARCHAR LIKE '%worldwide%'
ORDER BY s.EFFECTIVE_STATUS, s.NAME;
